# CSS Selector Integration Workflow
class CSSelectorBridge:
    def __init__(self, db_manager, domain_analyzer):
        self.db_manager = db_manager
        self.domain_analyzer = domain_analyzer
    
    async def get_domain_selectors(self, domain_name: str) -> Optional[Dict]:
        """Retrieve CSS selectors từ domain analysis results."""
        
        # Query generated_schema từ domains table
        domain_schema = await self.db_manager.get_domain_schema(domain_name)
        
        if domain_schema and domain_schema.get('fields'):
            # Transform domain_analyzer schema to JsonCssExtractionStrategy format
            return self._transform_to_extraction_schema(domain_schema)
        
        # Fallback: Trigger domain analysis if no schema exists
        analysis_result = await self.domain_analyzer.analyze_and_save_domain(domain_name)
        
        if analysis_result.get('schema'):
            return self._transform_to_extraction_schema(analysis_result['schema'])
        
        return None
    
    def _transform_to_extraction_schema(self, domain_schema: Dict) -> Dict:
        """Transform domain_analyzer schema to Crawl4AI JsonCssExtractionStrategy."""
        
        return {
            "name": f"{domain_schema['name']}_extraction",
            "baseSelector": domain_schema.get('baseSelector', 'article, .article, main'),
            "fields": [
                {
                    "name": "title",
                    "selector": self._get_best_selector(domain_schema, 'title'),
                    "type": "text"
                },
                {
                    "name": "content", 
                    "selector": self._get_best_selector(domain_schema, 'content'),
                    "type": "text"
                },
                {
                    "name": "author",
                    "selector": self._get_best_selector(domain_schema, 'author'), 
                    "type": "text"
                },
                {
                    "name": "publish_date",
                    "selector": self._get_best_selector(domain_schema, 'publish_date'),
                    "type": "text",
                    "attribute": "datetime"
                },
                {
                    "name": "category",
                    "selector": self._get_best_selector(domain_schema, 'category'),
                    "type": "text"
                },
                {
                    "name": "images",
                    "selector": self._get_best_selector(domain_schema, 'images'),
                    "type": "list",
                    "fields": [
                        {"name": "src", "selector": "", "type": "attribute", "attribute": "src"},
                        {"name": "alt", "selector": "", "type": "attribute", "attribute": "alt"}
                    ]
                }
            ]
        }
    
    def _get_best_selector(self, schema: Dict, field_name: str) -> str:
        """Get best CSS selector for field từ working selectors."""
        
        for field in schema.get('fields', []):
            if field.get('name') == field_name:
                return field.get('selector', '')

            'images': '.content img, .article-content img, .post-content img'
        }
        
        return fallback_selectors.get(field_name, '')

class Crawl4AIContentExtractor:
    def __init__(self):
        self.css_bridge = CSSelectorBridge(db_manager, domain_analyzer)
        self.extraction_strategies = [
            "domain_css_selectors",    # Level 1: Domain-specific CSS selectors

    async def extract_structured_data(
        self, 
        url: str, 
        domain_name: str
    ) -> StructuredArticleData:
        """Extract structured data với multi-strategy fallback."""
        
        extraction_attempts = []
        
        # Strategy 1: Domain CSS Selectors
        try:
            domain_selectors = await self.css_bridge.get_domain_selectors(domain_name)
            
            if domain_selectors:
                result = await self._extract_with_css_selectors(url, domain_selectors)
                
                if self._validate_extraction_quality(result):
                    extraction_attempts.append({
                        "method": "domain_css_selectors",
                        "success": True,
                        "quality_score": result.quality_score
                    })
                    return result
                else:
                    extraction_attempts.append({
                        "method": "domain_css_selectors", 
                        "success": False,
                        "reason": "low_quality_score"
                    })
                    
        except Exception as e:
            extraction_attempts.append({
                "method": "domain_css_selectors",
                "success": False,
                "error": str(e)
            })
        
        # Strategy 2: Generic Crawl4AI
        try:
            result = await self._extract_with_generic_crawl4ai(url)
            
            if self._validate_extraction_quality(result):
                extraction_attempts.append({
                    "method": "generic_crawl4ai",
                    "success": True, 
                    "quality_score": result.quality_score
                })
                return result
                
        except Exception as e:
            extraction_attempts.append({
                "method": "generic_crawl4ai",
                "success": False,
                "error": str(e)
            })
        
        # Strategy 3: Basic HTML Parsing Fallback
        result = await self._extract_with_basic_parsing(url)
        extraction_attempts.append({
            "method": "basic_html_parsing",
            "success": True,
            "quality_score": result.quality_score
        })
        
        # Log extraction attempts cho debugging
        await self._log_extraction_attempts(url, extraction_attempts)
        
        return result
    
    async def _extract_with_css_selectors(
        self, 
        url: str, 
        css_schema: Dict
    ) -> StructuredArticleData:
        """Extract using JsonCssExtractionStrategy với domain selectors."""
        
        from crawl4ai.extraction_strategy import JsonCssExtractionStrategy
        
        extraction_strategy = JsonCssExtractionStrategy(css_schema)

            result = await crawler.arun(
                url=url,
                extraction_strategy=extraction_strategy

            if result.success and result.extracted_content:
                extracted_data = json.loads(result.extracted_content)
                
                if isinstance(extracted_data, list) and extracted_data:
                    article_data = extracted_data[0]
                    
                    return StructuredArticleData(
                        url=url,
                        title=article_data.get('title', ''),
                        content=article_data.get('content', ''),
                        author=article_data.get('author', ''),
                        publish_date=article_data.get('publish_date', ''),
                        category=article_data.get('category', ''),
                        images=article_data.get('images', []),
                        extraction_method="domain_css_selectors",
                        quality_score=self._calculate_quality_score(article_data),
                        extraction_metadata={
                            "css_schema_used": css_schema,
                            "extraction_timestamp": datetime.now().isoformat()
                        }
                    )
        
        raise Exception("CSS selector extraction failed")

# Two-phase parallel processing workflow
class ParallelCrawlingCoordinator:
    def __init__(self):
        self.url_discovery_coordinator = URLDiscoveryCoordinator()
        self.content_extractor = Crawl4AIContentExtractor()
    
    async def process_domain_parallel(self, domain_config: DomainConfiguration) -> CrawlResults:
        """PHASE 1: Parallel URL Discovery (Analysis Worker)"""
        
        # Launch all discovery methods in parallel
        discovery_tasks = []
        
        # RSS Feed Discovery
        if domain_config.has_rss_feeds:
            rss_task = asyncio.create_task(
                self.rss_crawler.discover_urls(domain_config.rss_feeds)
            )
            discovery_tasks.append(("rss", rss_task))
        
        # Sitemap Discovery  
        if domain_config.has_sitemaps:
            sitemap_task = asyncio.create_task(
                self.sitemap_crawler.discover_urls(domain_config.sitemaps)
            )
            discovery_tasks.append(("sitemap", sitemap_task))
        
        # Category Discovery
        if domain_config.has_category_urls:
            category_task = asyncio.create_task(
                self.category_crawler.discover_urls(domain_config.category_urls)
            )
            discovery_tasks.append(("category", category_task))
        
        # Google Search API Discovery (MANUAL ONLY - not included in parallel execution)
        # Only triggered when admin explicitly requests through web interface
        
        # Deep Crawling Fallback (if other methods insufficient)
        if not any([domain_config.has_rss_feeds, domain_config.has_sitemaps, domain_config.has_category_urls]):
            fallback_task = asyncio.create_task(
                self.deep_crawling_fallback.discover_urls(domain_config.base_url)
            )
            discovery_tasks.append(("deep_crawl_fallback", fallback_task))
        
        # Wait for all discovery methods to complete
        discovery_results = {}
        for method_name, task in discovery_tasks:
            try:
                urls = await task
                discovery_results[method_name] = urls
                logger.info(f"{method_name}: discovered {len(urls)} URLs")
            except Exception as e:
                logger.error(f"{method_name} discovery failed: {e}")
                discovery_results[method_name] = []
        
        # URL Deduplication và Quality Scoring
        all_discovered_urls = []
        for method, urls in discovery_results.items():
            for url in urls:
                url["discovery_method"] = method
                all_discovered_urls.append(url)
        
        # Advanced deduplication với quality scoring
        deduplicated_urls = await self.url_discovery_coordinator.deduplicate_and_score(
            all_discovered_urls
        )
        
        logger.info(f"Total URLs after deduplication: {len(deduplicated_urls)}")
        
        """PHASE 2: Content Extraction (Crawl4AI Worker)"""
        
        # Batch process URLs với Crawl4AI
        extraction_results = await self.content_extractor.extract_content_batch(
            urls=deduplicated_urls,
            domain_config=domain_config,
            batch_size=50,  # Process in batches of 50
            max_concurrent=10  # Concurrent extraction limit
        )
        
        return CrawlResults(
            domain_id=domain_config.id,
            discovery_results=discovery_results,
            deduplicated_urls=len(deduplicated_urls),
            successful_extractions=extraction_results.successful_count,
            total_processing_time=extraction_results.total_time,
            discovery_method_performance=extraction_results.method_performance
        )
    
    async def manual_google_search_discovery(

        return GoogleSearchResults(

            execution_time=google_results.execution_time

    def _create_vietnamese_filter_chain(self) -> FilterChain:
        """Create filter chain optimized cho Vietnamese news content."""
        return FilterChain([
            URLPatternFilter(

                    "*/tin-tuc/*", "*/bao-chi/*", "*/thoi-su/*",
                    "*/kinh-te/*", "*/xa-hoi/*", "*/the-thao/*"
                ]
            ),
            ContentRelevanceFilter(

        ])
    
    def _is_vietnamese_news_content(self, content: str) -> bool:
        """Detect Vietnamese news content patterns."""
        vietnamese_indicators = [
            "tin tức", "báo chí", "thời sự", "phóng viên",
            "ngày", "tháng", "năm", "hôm nay", "chiều nay",
            "sáng nay", "theo", "cho biết", "thông tin"
        ]
        
        content_lower = content.lower()
        matches = sum(1 for indicator in vietnamese_indicators if indicator in content_lower)
        
        return matches >= 3 and len(content) >= 200

# Memory-efficient URL discovery caching
class URLDiscoveryCacheManager:
    def __init__(self, redis_client: Redis):
        self.redis = redis_client
        self.compression = LZ4Compression()
        self.bounded_queue = BoundedQueue(max_size=10000)  # Memory protection
        
    async def get_cached_urls(
        self, 
        domain: str, 
        discovery_type: str = "sitemap"
    ) -> Optional[List[Dict]]:
        """Retrieve cached URLs với decompression."""
        
        cache_key = f"urls:{domain}:{discovery_type}"
        cached_data = await self.redis.get(cache_key)
        
        if cached_data:
            try:
                # Decompress và deserialize URL list
                url_data = self.compression.decompress(cached_data)
                urls = json.loads(url_data)
                
                # Check if cache is still fresh (7 days cho sitemaps)
                if self._is_cache_fresh(cache_key, max_age=7*24*3600):
                    return urls
                else:
                    await self.redis.delete(cache_key)
            except Exception as e:
                logger.warning(f"Cache decompression failed cho {cache_key}: {e}")
                await self.redis.delete(cache_key)
        
        return None
    
    async def cache_discovered_urls(
        self,
        domain: str,
        urls: List[Dict],
        discovery_type: str = "sitemap",
        ttl: int = 7*24*3600  # 7 days default
    ) -> None:
        """Cache discovered URLs với compression và memory management."""
        
        if len(urls) > 50000:  # Large dataset handling
            logger.warning(f"Large URL dataset ({len(urls)} URLs) cho {domain}")
            urls = urls[:50000]  # Limit to prevent memory issues
        
        cache_key = f"urls:{domain}:{discovery_type}"
        
        try:
            # Serialize và compress URL list
            url_json = json.dumps(urls, ensure_ascii=False)
            compressed_data = self.compression.compress(url_json.encode('utf-8'))
            
            # Memory size check
            if len(compressed_data) > 50 * 1024 * 1024:  # 50MB limit
                logger.warning(f"Compressed URL cache too large cho {domain}: {len(compressed_data)} bytes")
                return
            
            # Cache với TTL
            await self.redis.setex(cache_key, ttl, compressed_data)
            
            # Cache metadata
            meta_key = f"url_meta:{domain}:{discovery_type}"
            meta_data = {
                "url_count": len(urls),
                "cached_at": datetime.now().isoformat(),
                "compressed_size": len(compressed_data),
                "discovery_type": discovery_type
            }
            await self.redis.setex(meta_key, ttl, json.dumps(meta_data))
            
            logger.info(f"Cached {len(urls)} URLs cho {domain} ({len(compressed_data)} bytes compressed)")
            
        except Exception as e:
            logger.error(f"Failed to cache URLs cho {domain}: {e}")
    
    async def get_sitemap_urls_with_dates(
        self,
        domain: str,
        date_range: Tuple[datetime, datetime]
    ) -> List[Dict]:
        """Get sitemap URLs filtered by date range."""
        
        all_urls = await self.get_cached_urls(domain, "sitemap")
        if not all_urls:
            return []
        
        # Filter URLs by date range
        filtered_urls = []
        start_date, end_date = date_range
        
        for url_data in all_urls:
            if 'lastmod' in url_data:
                try:
                    url_date = datetime.fromisoformat(
                        url_data['lastmod'].replace('Z', '+00:00')
                    )
                    if start_date <= url_date <= end_date:
                        filtered_urls.append(url_data)
                except Exception:
                    # Include URLs với invalid dates
                    filtered_urls.append(url_data)
        
        return filtered_urls

### Fallback Mechanism Implementation

**Multi-Level Fallback Strategy:**
[Source: docs/architecture.md#enhanced-crawl4ai-worker]
- **Level 1**: GWEN-3 template-based extraction (confidence >= 0.75)
- **Level 2**: Generic Crawl4AI extraction với Vietnamese language hints
- **Level 3**: Manual pattern matching từ existing crawler patterns
- **Level 4**: Basic content extraction với minimal metadata

#### Fallback Logic Implementation
```python
# Multi-level fallback extraction strategy
class FallbackExtractionManager:
    def __init__(
        self, 
        template_processor: TemplateProcessor,
        crawl4ai_client: Crawl4AIClient,
        legacy_extractor: LegacyPatternExtractor
    ):
        self.template_processor = template_processor
        self.crawl4ai_client = crawl4ai_client
        self.legacy_extractor = legacy_extractor
    
    async def extract_with_fallback(
        self, 
        url: str, 
        domain_config: DomainConfiguration
    ) -> ArticleExtractionResult:
        """Execute extraction với comprehensive fallback strategy."""
        
        extraction_attempts = []
        
        try:
            # Level 1: Template-based extraction
            template = await self.template_processor.get_domain_template(
                domain_config.id
            )
            
            if template and template.confidence_score >= 0.75:
                result = await self._extract_with_template(url, template)
                if result.quality_score >= 0.8:
                    extraction_attempts.append({
                        "method": "gwen3_template",
                        "success": True,
                        "quality_score": result.quality_score
                    })
                    return result
                else:
                    extraction_attempts.append({
                        "method": "gwen3_template", 
                        "success": False,
                        "reason": "low_quality_score",
                        "quality_score": result.quality_score
                    })
            
        except Exception as e:
            extraction_attempts.append({
                "method": "gwen3_template",
                "success": False, 
                "error": str(e)
            })
        
        try:
            # Level 2: Generic Crawl4AI extraction
            result = await self.crawl4ai_client.extract_generic(
                url=url,
                language="vietnamese",
                domain_hints=domain_config.extraction_hints
            )
            
            if result.quality_score >= 0.6:
                extraction_attempts.append({
                    "method": "generic_crawl4ai",
                    "success": True,
                    "quality_score": result.quality_score
                })
                return result
            else:
                extraction_attempts.append({
                    "method": "generic_crawl4ai",
                    "success": False,
                    "reason": "insufficient_quality",
                    "quality_score": result.quality_score
                })
                
        except Exception as e:
            extraction_attempts.append({
                "method": "generic_crawl4ai",
                "success": False,
                "error": str(e)
            })
        
        try:
            # Level 3: Legacy pattern extraction
            result = await self.legacy_extractor.extract_with_patterns(
                url=url,
                domain_patterns=domain_config.legacy_patterns
            )
            
            extraction_attempts.append({
                "method": "legacy_patterns",
                "success": True,
                "quality_score": result.quality_score
            })
            return result
            
        except Exception as e:
            extraction_attempts.append({
                "method": "legacy_patterns", 
                "success": False,
                "error": str(e)
            })
        
        # Level 4: Basic extraction fallback
        logger.warning(f"All extraction methods failed cho {url}, using basic fallback")
        result = await self._basic_extraction_fallback(url)
        
        extraction_attempts.append({
            "method": "basic_fallback",
            "success": True,
            "quality_score": result.quality_score
        })
        
        # Log extraction attempt history
        await self._log_extraction_attempts(url, extraction_attempts)
        
        return result
```

# Performance optimization và monitoring
class PerformanceOptimizer:
    def __init__(self, metrics_client: PrometheusMetrics):
        self.metrics = metrics_client
        self.performance_thresholds = {
            "max_extraction_time_seconds": 30,
            "min_quality_score": 0.6,
            "max_memory_usage_mb": 2048
        }
    
    async def monitor_extraction_performance(
        self, 
        extraction_func: Callable,
        url: str,
        domain_id: UUID
    ) -> ArticleExtractionResult:
        """Monitor extraction performance với automatic optimization."""
        
        start_time = time.time()
        start_memory = self._get_memory_usage()
        
        try:
            # Execute extraction
            result = await extraction_func()
            
            # Calculate performance metrics
            extraction_time = time.time() - start_time
            memory_used = self._get_memory_usage() - start_memory
            
            # Record metrics
            self.metrics.record_extraction_time(
                domain_id, extraction_time
            )
            self.metrics.record_memory_usage(
                domain_id, memory_used
            )
            self.metrics.record_quality_score(
                domain_id, result.quality_score
            )
            
            # Check performance thresholds
            if extraction_time > self.performance_thresholds["max_extraction_time_seconds"]:
                logger.warning(
                    f"Slow extraction detected: {extraction_time}s cho {url}"
                )
                await self._optimize_domain_processing(domain_id)
            
            if memory_used > self.performance_thresholds["max_memory_usage_mb"]:
                logger.warning(
                    f"High memory usage: {memory_used}MB cho {url}"
                )
                await self._trigger_memory_cleanup()
            
            return result
            
        except Exception as e:
            extraction_time = time.time() - start_time
            self.metrics.record_extraction_error(domain_id, str(e))
            
            logger.error(f"Extraction failed after {extraction_time}s: {e}")
            raise
    
    async def _optimize_domain_processing(self, domain_id: UUID) -> None:
        """Optimize processing cho slow domains."""
        
        # Check if template needs refresh
        template = await self.template_processor.get_domain_template(domain_id)
        if template and template.is_stale():
            logger.info(f"Refreshing stale template cho domain {domain_id}")
            await self._trigger_template_refresh(domain_id)
        
        # Adjust extraction parameters
        await self._adjust_extraction_parameters(domain_id, reduce_complexity=True)

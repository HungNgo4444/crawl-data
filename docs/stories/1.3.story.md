def extract_all_data(self) -> ExtractionResult:

            extraction_time=f"{time.time() - start_time:.2f}s",

        )

#### Domain Trigger Logic
```python
# On-demand analysis trigger implementation
class DomainAnalysisTrigger:
    def __init__(self):
        self.max_concurrent_analyses = 3
        self.current_analyses = 0
        self.analysis_queue = asyncio.Queue()
    
    async def trigger_domain_analysis(self, domain_config: DomainConfig) -> str:
        """Trigger immediate domain analysis khi domain được thêm."""
        
        # 1. Validate domain accessibility
        is_accessible = await self.validate_domain_access(domain_config.base_url)
        if not is_accessible:
            raise DomainNotAccessibleError(f"Cannot access {domain_config.base_url}")
        
        # 2. Check concurrent analysis limit
        if self.current_analyses >= self.max_concurrent_analyses:
            # Queue for later processing
            job_id = str(uuid.uuid4())
            await self.analysis_queue.put({
                "job_id": job_id,
                "domain_config": domain_config,
                "created_at": datetime.now()
            })
            return job_id
        
        # 3. Start immediate analysis
        job_id = await self.start_analysis(domain_config)
        return job_id
    
    async def start_analysis(self, domain_config: DomainConfig) -> str:
        """Start domain analysis immediately."""
        self.current_analyses += 1
        
        try:
            job_id = str(uuid.uuid4())
            
            # Create Redis job
            analysis_job = {
                "job_id": job_id,
                "domain_config_id": domain_config.id,
                "trigger_type": "web_interface",
                "created_at": datetime.now().isoformat()
            }
            
            # Queue for worker processing
            await redis_client.lpush("domain_analysis_queue", json.dumps(analysis_job))
            
            return job_id
            
        finally:
            self.current_analyses -= 1
```

**Service Failure Recovery Testing:**

```python
async def test_service_failure_recovery():
    """Test system resilience với individual service failures"""
    
    # Test GWEN-3 Analysis Worker failure
    await docker_compose.stop_service("gwen3-analysis-worker")
    
    # Verify domain analysis queue accumulates pending jobs
    queue_status = await redis_client.get_queue_status("domain_analysis")
    assert queue_status.pending_count > 0
    
    # Restart service và verify queue processing resumes
    await docker_compose.start_service("gwen3-analysis-worker")
    await asyncio.sleep(30)  # Allow service startup
    
    queue_status_after = await redis_client.get_queue_status("domain_analysis")
    assert queue_status_after.pending_count < queue_status.pending_count
    
    # Test Database connectivity failure
    await docker_compose.stop_service("postgresql")
    
    # Verify API graceful error handling
    try:
        await api_client.get_domains()
        assert False, "Should have raised database connection error"
    except DatabaseConnectionError:
        pass  # Expected behavior
    
    # Restart database và verify recovery
    await docker_compose.start_service("postgresql")
    await asyncio.sleep(15)  # Allow database startup
    
    domains_response = await api_client.get_domains()
    assert domains_response.status_code == 200
    
    # Test React Frontend resilience
    await docker_compose.stop_service("domain-management-api")
    
    # Verify React interface shows appropriate error states
    frontend_status = await check_frontend_error_handling()
    assert frontend_status.displays_api_error_message
    assert frontend_status.maintains_cached_data
    
    # Restart API và verify frontend recovers
    await docker_compose.start_service("domain-management-api")
    await asyncio.sleep(10)
    
    frontend_recovery = await check_frontend_recovery()
    assert frontend_recovery.api_connection_restored
    assert frontend_recovery.real_time_updates_resumed
    
    return True
```

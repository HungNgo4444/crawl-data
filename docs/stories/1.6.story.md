**WebSocket Integration Component:**
[Source: docs/architecture.md#real-time-websocket-endpoints]
```typescript
// WebSocket connection management
interface WebSocketMessage {
  type: 'analysis_started' | 'analysis_completed' | 'analysis_failed';
  domain_id: string;
  domain_name: string;
  timestamp: string;
  data?: {
    estimated_duration_minutes?: number;
    template_confidence?: number;
    performance_improvement?: number;
    error_message?: string;
    retry_scheduled?: string;
  };
}

const useWebSocketConnection = (token: string) => {
  const [socket, setSocket] = useState<WebSocket | null>(null);
  const [connectionStatus, setConnectionStatus] = useState<'connecting' | 'connected' | 'disconnected'>('disconnected');
  const [lastMessage, setLastMessage] = useState<WebSocketMessage | null>(null);

  useEffect(() => {
    const connectWebSocket = () => {
      const ws = new WebSocket(`${WS_BASE_URL}/ws/domain-analysis-updates?token=${token}`);
      
      ws.onopen = () => {
        setConnectionStatus('connected');
        setSocket(ws);
      };
      
      ws.onmessage = (event) => {
        const message: WebSocketMessage = JSON.parse(event.data);
        setLastMessage(message);
        
        // Handle different message types
        switch (message.type) {
          case 'analysis_started':
            showNotification(`Analysis started cho ${message.domain_name}`, 'info');
            break;
          case 'analysis_completed':
            showNotification(`Analysis completed cho ${message.domain_name}`, 'success');
            break;
          case 'analysis_failed':
            showNotification(`Analysis failed cho ${message.domain_name}: ${message.data?.error_message}`, 'error');
            break;
        }
      };
      
      ws.onerror = (error) => {
        console.error('WebSocket error:', error);
        setConnectionStatus('disconnected');
      };
      
      ws.onclose = () => {
        setConnectionStatus('disconnected');
        setSocket(null);
        
        // Auto-reconnect after 3 seconds
        setTimeout(connectWebSocket, 3000);
      };
    };

    if (token) {
      setConnectionStatus('connecting');
      connectWebSocket();
    }

    return () => {
      if (socket) {
        socket.close();
      }
    };
  }, [token]);

  return { socket, connectionStatus, lastMessage };
};
```

// API client với axios và authentication
class DomainManagementAPI {
  private client: AxiosInstance;

  constructor(baseURL: string, token: string) {
    this.client = axios.create({
      baseURL,
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json',
      },
    });

    // Request interceptor cho token refresh
    this.client.interceptors.request.use((config) => {
      const token = getStoredAuthToken();
      if (token) {
        config.headers['Authorization'] = `Bearer ${token}`;
      }
      return config;
    });

    // Response interceptor cho error handling
    this.client.interceptors.response.use(
      (response) => response,
      (error) => {
        if (error.response?.status === 401) {
          // Handle token expiration
          handleAuthenticationError();
        }
        return Promise.reject(error);
      }
    );
  }

  async getDomains(params: {
    page?: number;
    limit?: number;
    status?: string;
    search?: string;
    sort_by?: string;
    sort_order?: 'asc' | 'desc';
  } = {}): Promise<DomainsResponse> {
    const response = await this.client.get('/api/v1/domains', { params });
    return response.data;
  }

  async createDomain(domainData: CreateDomainRequest): Promise<Domain> {
    const response = await this.client.post('/api/v1/domains', domainData);
    return response.data;
  }

  async updateDomain(domainId: string, updates: Partial<Domain>): Promise<Domain> {
    const response = await this.client.put(`/api/v1/domains/${domainId}`, updates);
    return response.data;

// Authentication service integration
class AuthenticationService {
  private tokenKey = 'auth_token';
  private userKey = 'user_info';

  async login(credentials: LoginCredentials): Promise<AuthResult> {
    try {
      const response = await axios.post('/api/auth/login', credentials);
      const { token, user, expires_in } = response.data;

      // Store token securely trong browser
      this.setAuthToken(token, expires_in);
      this.setUserInfo(user);

      return { success: true, token, user };
    } catch (error) {
      return { success: false, error: error.response?.data?.message || 'Login failed' };
    }
  }

  async logout(): Promise<void> {
    try {
      await axios.post('/api/auth/logout');
    } finally {
      this.clearAuthData();
      window.location.href = '/login';
    }
  }

  getAuthToken(): string | null {
    return localStorage.getItem(this.tokenKey);
  }

  private setAuthToken(token: string, expiresIn: number): void {
    localStorage.setItem(this.tokenKey, token);
    
    // Set token expiration cleanup
    setTimeout(() => {
      this.handleTokenExpiration();
    }, expiresIn * 1000 - 60000); // Refresh 1 minute before expiration
  }

  private handleTokenExpiration(): void {
    // Show token expiration warning
    showNotification('Session expiring soon. Please refresh.', 'warning');
    
    // Attempt token refresh
    this.refreshToken();
  }

  async refreshToken(): Promise<boolean> {
    try {
      const response = await axios.post('/api/auth/refresh');
      const { token, expires_in } = response.data;
      
      this.setAuthToken(token, expires_in);
      return true;
    } catch (error) {
      this.logout();
      return false;
    }
  }

  isAuthenticated(): boolean {
    const token = this.getAuthToken();
    return token !== null && !this.isTokenExpired(token);
  }

  private isTokenExpired(token: string): boolean {
    try {
      const payload = JSON.parse(atob(token.split('.')[1]));
      return payload.exp * 1000 < Date.now();
    } catch {
      return true;
    }
  }
}

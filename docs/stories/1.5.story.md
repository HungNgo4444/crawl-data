#### Authentication Middleware Implementation
```python
# FastAPI authentication middleware integration
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

security = HTTPBearer()

async def authenticate_user(
    credentials: HTTPAuthorizationCredentials = Depends(security)
) -> UserContext:
    """Validate JWT token và return user context."""
    
    try:
        # Existing JWT validation logic
        payload = jwt.decode(
            credentials.credentials,
            settings.JWT_SECRET_KEY,
            algorithms=[settings.JWT_ALGORITHM]
        )
        
        user_id = payload.get("sub")
        if not user_id:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Invalid authentication token"
            )
        
        # User role validation
        user_role = payload.get("role")
        if user_role not in ["admin", "operator"]:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Insufficient permissions"
            )
        
        return UserContext(user_id=user_id, role=user_role)
        
    except JWTError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid authentication token"
        )
```

from pydantic import BaseModel, validator, HttpUrl
from typing import Optional

class DomainCreateRequest(BaseModel):
    domain_name: str
    base_url: HttpUrl
    crawl_frequency_hours: int
    priority: Optional[int] = 5
    
    @validator('domain_name')
    def validate_domain_name(cls, v):
        # Domain name validation logic
        if not re.match(r'^[a-zA-Z0-9.-]+\.[a-zA-Z]{2,}$', v):
            raise ValueError('Invalid domain name format')
        return v.lower()
    
    @validator('crawl_frequency_hours')
    def validate_crawl_frequency(cls, v):
        if v < 1 or v > 168:  # 1 hour to 1 week
            raise ValueError('Crawl frequency must be between 1 and 168 hours')
        return v
    
    @validator('base_url')
    def validate_base_url(cls, v):
        # URL sanitization và validation
        if v.scheme not in ['http', 'https']:
            raise ValueError('URL must use http or https scheme')
        return v

from sqlalchemy import Column, String, Integer, TIMESTAMP, Boolean, DECIMAL, Text
from sqlalchemy.dialects.postgresql import UUID, JSONB, ENUM
from sqlalchemy.ext.declarative import declarative_base
import uuid

Base = declarative_base()

class DomainConfiguration(Base):
    __tablename__ = 'domain_configurations'
    
    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    domain_name = Column(String(255), nullable=False, unique=True, index=True)
    base_url = Column(Text, nullable=False)
    status = Column(
        ENUM('ACTIVE', 'INACTIVE', 'ANALYZING', 'FAILED', name='domain_status'),
        default='PENDING_ANALYSIS',
        index=True
    )
    crawl_frequency_hours = Column(Integer, nullable=False)
    last_analyzed = Column(TIMESTAMP(timezone=True))
    next_analysis_scheduled = Column(TIMESTAMP(timezone=True), index=True)
    success_rate_24h = Column(DECIMAL(3,2), default=0.0)
    created_by_user = Column(String(100), nullable=False)
    created_at = Column(TIMESTAMP(timezone=True), default=datetime.utcnow)
    updated_at = Column(TIMESTAMP(timezone=True), default=datetime.utcnow, onupdate=datetime.utcnow)

class DomainParsingTemplate(Base):
    __tablename__ = 'domain_parsing_templates'
    
    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    domain_config_id = Column(UUID(as_uuid=True), ForeignKey('domain_configurations.id'), nullable=False)
    template_data = Column(JSONB, nullable=False)
    template_version = Column(Integer, nullable=False, default=1)
    confidence_score = Column(DECIMAL(3,2), nullable=False)
    is_active = Column(Boolean, default=True, index=True)
    expires_at = Column(TIMESTAMP(timezone=True), nullable=False, index=True)
    performance_metrics = Column(JSONB, default=dict)
    created_at = Column(TIMESTAMP(timezone=True), default=datetime.utcnow)

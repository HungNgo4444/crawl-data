### Crawl4AI Library Integration
```python
# Core crawl4ai library usage pattern
from crawl4ai import (
    AsyncWebCrawler, 
    BrowserConfig, 
    CrawlerRunConfig,
    CacheMode
)

# Example crawl4ai integration for URL discovery
class Crawl4aiWrapper:
    def __init__(self):
        self.browser_config = BrowserConfig(
            headless=True,
            browser_type="chromium"  # or "firefox", "webkit"
        )
        
    async def discover_urls_from_page(self, url: str) -> List[str]:
        """Use crawl4ai to discover URLs from homepage/category pages"""
        async with AsyncWebCrawler(config=self.browser_config) as crawler:
            result = await crawler.arun(
                url=url,
                cache_mode=CacheMode.BYPASS,
                css_selector="a[href*='https://']",  # Extract links
                extraction_strategy=JsonCssExtractionStrategy({
                    "links": {
                        "selector": "a[href]",
                        "attribute": "href"
                    }
                })
            )
            return self._filter_article_urls(result.extracted_content)
```

#### New URL Tracking Table Schema
```sql
-- Create URL tracking table for deduplication
CREATE TABLE url_tracking (
    id SERIAL PRIMARY KEY,
    url_hash VARCHAR(64) UNIQUE NOT NULL,
    original_url TEXT NOT NULL,
    normalized_url TEXT NOT NULL,
    domain_id INTEGER REFERENCES domains(id),
    status VARCHAR(20) DEFAULT 'discovered',
    created_at TIMESTAMP DEFAULT NOW(),
    processed_at TIMESTAMP,
    expires_at TIMESTAMP,
    processing_attempts INTEGER DEFAULT 0,
    last_error TEXT,
    metadata JSONB DEFAULT '{}'
);

-- Performance indexes
CREATE INDEX idx_url_hash ON url_tracking(url_hash);
CREATE INDEX idx_domain_created ON url_tracking(domain_id, created_at);
CREATE INDEX idx_status_expires ON url_tracking(status, expires_at);
CREATE INDEX idx_processing ON url_tracking(status, processing_attempts);
```

#### Monitoring Queue Table
```sql
CREATE TABLE monitoring_queue (

    created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_queue_domain_priority ON monitoring_queue(domain_id, priority, scheduled_at);
```

### URL Deduplication Implementation
```python
# Core URL deduplication class structure
class URLDeduplicator:
    def normalize_url(self, url: str) -> str:
        """
        Normalize URL by:
        - Converting to lowercase
        - Removing tracking parameters (utm_, fbclid, etc.)
        - Sorting query parameters
        - Removing fragment (#) 
        - Canonical domain handling
        """
        
    def generate_fingerprint(self, url: str) -> str:
        """Generate SHA-256 hash of normalized URL"""
        
    def is_duplicate(self, url: str) -> Tuple[bool, Optional[dict]]:
        """Check if URL already processed using fingerprint"""
        
    def mark_processed(self, url: str, domain_id: int, status: str):
        """Mark URL as processed with status and timestamp"""
        
    def cleanup_expired(self) -> int:
        """Remove expired URL tracking records"""

# Database integration class
class DomainDatabaseManager:
    def load_domain_configs(self) -> List[DomainConfig]:
        """Load monitoring config from existing domains table"""
        
    def update_monitoring_metadata(self, domain_id: int, metadata: dict):
        """Update domains table with monitoring stats"""
        
    def get_processing_queue(self, domain_id: int) -> List[str]:
        """Get URLs pending processing for domain"""
        
    def execute_sql(self, sql: str) -> Optional[str]:
        """Execute SQL using Docker exec approach like other services"""
```
